    # git 命令

    # 配置

    命令 作用
    git config --global user.name 用户名 配置用户姓名
    git config --global user.email 邮箱地址 配置用户地址
    git config --list 查看配置列表

    # 添加

    命令 作用
    git init 创建本地仓库
    git add 文件名 添加文件/修改至暂存区
    git commit -m '提交说明' 将暂存区的文件提交到本地仓库
    git commit --all '提交说明' 添加文件至暂存区并提交到本地仓库
    git stash 暂存所有修改, 清空工作区
    git stash pop 恢复上次暂存的所有修改至工作区

    # 查看

    命令 作用
    git status 查看工作区文件状态
    git log 查看提交操作日志
    git log --oneline 查看提交操作日志, 并使其一行显示
    git reflog 查看所有操作日志

    # 撤消

    命令 作用
    git checkout 文件名 暂存区文件覆盖工作区文件
    git rm --cached 文件名 删除暂存区文件
    git reset --hard 提交ID 重置到任意一次提交状态

    # 分支

    命令 作用
    git branch 查看所有分支, 及当前所处分支
    git branch 分支名 创建分支
    git checkout 分支名 切换分支
    git checkout -d 分支名 创建并切换至该分支
    git merge 分支名 合并分支
    git branch -d 分支名 删除分支, 没有合并的分支不可删除
    git branch -D 分支名 强制删除分支

    # 仓库

    命令 作用
    git push 远程仓库地址/远程仓库地址别名 分支名 将本地仓库推送至远程仓库的对应分支
    git push -u 远程仓库地址别名 分支名 使用别名推送, 并使后续可用 git push 直接推送
    git remote add 远程仓库地址别名 远程仓库地址 创建远程仓库地址别名
    git remote remove 远程仓库地址别名 删除远程仓库地址别名
    git clone 远程仓库地址 将远程仓库整个克隆到本地
    git pull 远程仓库地址 分支名 将远程仓库的最新版本覆盖本地

    # 密钥

    命令 作用
    ssh-keygen 生成 ssh 密钥文件
    ssh-keygen -t rsa -C 注册邮箱 生成 ssh 密钥文件, 并绑定邮箱
