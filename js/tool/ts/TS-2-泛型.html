# 泛型

  1. 泛型是对数据类型的扩展, 使其不再具有固定类型
  2. 在函数调用/类实例化/对接接口的同时指定具体类型, 也可以不指定自动推断
  3. 避免重复代码

# 泛型函数

  1. 在函数定义的同时指定泛型数据的位置, <T> 放在函数名后
  2. 在函数调用的同时指定泛型的具体类型, <具体类型> 放在函数名后
  3. 在函数调用的同时也可以不指定泛型的具体类型, 此时数据类型会自动推断
function getData<T>(chunk:T):number {
    if (typeof chunk == 'string') {
        return chunk.length;
    } 
    if (typeof chunk == 'number') {
        return chunk;
    }
    return -1;
}
console.log(getData<string>('pikachu')); // 函数调用, 指明类型

# 泛型类

  1. 在声明类的同时指定泛型数据的位置, <T> 放在类名后
  2. 在实例化的同时指定泛型的具体类型, <具体类型> 放在类名后
  3. 在函数调用的同时也可以不指定泛型的具体类型, 此时数据类型会自动推断
class Handler<T> {
    dataArr:T[] = [];
    constructor() {

    };
    addData(param:T):void {
        this.dataArr.push(param);
    }
    sortArr() {
        return this.dataArr.sort();
    }
}
let h = new Handler<number>(); // 类实例化, 指明类型
h.addData(24324);
h.addData(63578);
h.addData(23865);
h.addData(24376);
h.addData(24334);
console.log(h.sortArr())

# 泛型接口

  1. 泛型接口是对一般接口的扩展
  2. 在设置接口的同时指定泛型数据的位置, <T> 放在接口名后
  3. 在对接接口的同时指定泛型的具体类型, <具体类型> 放在接口名后
// 1. 属性泛型接口
interface Rich<T> {
    money:T;
    weight:T
}
function identify(person:Rich<number>):boolean { // 对接接口, 指明类型
    if (person.money > 10 && person.weight > 200) {
        return true;
    }   
    return false;
}
console.log(identify({
    money: 5,
    weight: 230
}));

// 2. 函数类型泛型接口
interface dataType<T> {
    (v1:T, v2:T):boolean
}
let programmer:dataType<number> = function (gz, hair) { // 对接接口, 指明类型
    if (gz > 10000 && hair < 1) {
        return true;
    }
    return false;
};
console.log(programmer(8000, 0.2))

# 泛型约束

function getId<T>(arg:T):boolean {
    if (arg.length > 9) { // arg.length 会报错
        return true;
    }
    return false;
}
  1. 泛型约束是对泛型的一个补充, 使泛型数据具有一定的属性
  2. 泛型约束是通过接口 + extends 关键字实现的
interface argInter {
    length:number;
}
function getId<T extends argInter>(arg:T):boolean {
    if (arg.length > 9) {
        return true;
    }
    return false;
}
console.log(getId('hello typescript'));
