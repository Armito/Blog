# 接口

# 一. 属性接口

  1. 属性接口是对 JSON 格式的数据规范
  2. 属性接口中的属性均为必传属性; 设定为可选属性后, 则可以不传
  3. 直接传入数据时, 有多余的属性会报错; 赋予变量后传入, 多余的属性不会报错
interface stature {
    face: boolean;
    body?: boolean;
    height: number
}
function mate(girl:stature):void {
    if (girl.height > 160) {
        if (girl.face) {
            if (girl.body) {
                console.log('ok, let\'s mate');
                return;
            }
            console.log('I\'ll have a try');
            return
        }
        console.log('I\'ll have a see');
        return
    }
    console.log('No way!');
}
mate({
    height: 180, 
    face: true
});

# 二. 函数类型接口

  1. 函数类型的接口是对匿名函数的规范
  2. 形参的名字可以与接口不一样, 但如果参数是引用类型, 其键名必须相同
interface matefunc {
    (param:stature, sex?:string):void
}
let mating:matefunc = function (girl, gender='female') {
    if (gender == 'female') {
        if (girl.height > 160) {
            if (girl.face) {
                if (girl.body) {
                    console.log('ok, let\'s mate');
                    return;
                }
                console.log('I\'ll have a try');
                return
            }
            console.log('I\'ll have a see');
            return
        }
        console.log('That is a pity!');
        return
    }
    console.log('No way!')
};
mating({
    height: 150,
    face: true,
    body: true,
});

# 三. 可索引类型的接口

  1. 可索引类型接口是对数组, 对象的格式数据的规范
  2. 可索引类型接口可以对键名的数据类型进行规范

# 四. 类类型接口

  1. 类类型接口是对类的格式数据进行规范
  2. 类类型接口中, 可以有属性, 也可以有方法
  3. 类类型接口中的属性和方法均为必填属性
  4. 类中的属性必须在 constructor 函数之外定义
  5. 类中的方法可以没有参数, 有参数必须符合接口规定
  6. 类使用 implements 关键字对接接口
interface Personconfig {
    name:string;
    eat(food:string):void;
}
class Person implements Personconfig {
    name:string;
    constructor(name:string) {
        this.name = name;
    }
    eat():void{
        console.log(2333)
    }
}
let p = new Person('pikachu');
p.eat();

# 五. 类作为接口使用

  1. 类可以作为属性接口直接使用
  2. 作为接口的类中的实例属性需要增加 | undefined 代替初始化
  3. 使用方法与属性接口相同
class Article {
    title:string | undefined;
    author:string | undefined;
}
function publish(post:Article):boolean {
    if (post.author && post.title) {
        return true;
    }
    return false;
}
console.log(publish({
    title: undefined,
    author: 'ppp',
}))

# 六. 接口的继承

  1. 接口之间可以使用 extends 关键字实现继承
  2. 子类继承父类后依然可以直接使用 implements 关键字对接接口
class F {
    constructor(name:string) {
        this.name = name
    }
    name:string
}
class S extends F implements Personconfig {
    constructor(name:string) {
        super(name);
    }
    eat():void {
        console.log(233)
    }
}
let s = new S('pikachu');
s.eat();
