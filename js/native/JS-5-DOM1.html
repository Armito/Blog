# DOM1

# 一. Node节点

# 1.1 节点类型

Node.ELEMENT_NODE: 1
Node.ATTRIBUTE_NODE: 2
Node.TEXT_NODE: 3
Node.CDATA_SECTION_NODE: 4
Node.ENTITY_REFERENCE_NODE: 5
Node.ENTITY_NODE: 6
Node.PROCESSING_INSTRUCTION_NODE: 7
Node.COMMENT_NODE: 8
Node.DOCUMENT_NODE: 9
Node.DOCUMENT_TYPE_NODE: 10
Node.DOCUMENT_FRAGMENT_NODE: 11
Node.NOTATION_NODE: 12

# 1.2 节点属性

// nodeType 节点类型属性，值为 1~12 数字
someNode.nodeType
// nodeName 节点名称属性，保存的始终都是节点的名称
someNode.nodeName
// nodeValue 节点值属性
someNode.nodeValue

# 1.3 节点关系

  • 子节点
// childNodes 子节点属性，包含一个 NodeList 对象
someNode.childNodes
// 1.下标获取节点
someNode.childNodes[i]
// 2.item() 方法获取节点
someNode.childNodes.item(i)
// 3.将 NodeList 转化为真数组
Array.prototype.slice.call(someNode.childNodes, 0)

// 首个和最后一个子节点属性 firstChild, lastChild
someNode.firstChild
someNode.lastChild
  • 父节点与兄弟节点
// parentNode 父节点属性
someNode.parentNode
// previousSibling, nextSibling 兄弟节点属性
someNode.previousSibling
someNode.nextSibling
  • 关系属性和方法
// ownerDocument 属性，该属性永远指向表示整个文档的文档节点 document
someNode.ownerDocument
// hasChildNodes() 方法，该方法在节点包含一或多个子节点的情况下返回 true
someNode.hasChildNodes()

DOM扩展

// 检测的节点是后代节点，该方法返回 true；否则，返回 false
someNode.contains(someNode)

# 1.4 节点操作

  • 子节点操作
// 增, 返回新增节点
someNode.appendChild(newChildNode)
// 如果传入到 appendChild()中的节点已经是文档的一部分，那结果就是将该节点从原来的位置转移到新位置
someNode.insertBefore(newChildNode, someNode.childNode)
// 改，返回被替换的节点
someNode.replaceChild(newChildNode, someNode.childNode)
// 删，返回删除的节点
someNode.removeChild(someNode.childNode)
  • 其他节点操作
// 复制节点，参数为一个布尔值，true: 深拷贝， false: 浅拷贝
someNode.cloneNode(Boolean)

# 二. Document类型节点

Node => Document => HTMLDocument => document节点

# 2.1 document节点属性

document.nodeType：9
document.nodeName：'#document'
document.nodeValue：null
document.parentNode：null
document.ownerDocument：null

# 2.2 document节点的子节点

// 获取 DocumentType 节点
document.doctype
// 获取 html 元素节点
document.documentElement
// 获取 body 元素节点
document.body

DOM扩展

// 获取 head 元素节点
document.head

# 2.3 document节点的文档信息

// 获取文档标题
document.title
// 获取完整 url
document.URL
// 获取域名
document.domain
// 获取来源页面的 url
document.referrer

DOM扩展

document.readyState
// loading，文档正在加载；complete，文档已加载完
document.compatMode
// 渲染页面的模式：CSS1Compat，标准模式；BackCompat，混杂模式
document.charset
// 文档中实际使用的字符集，也可以用来指定新字符集
document.defaultCharset
// 当前文档默认的字符集

# 2.4 document节点文档的元素查找

  • 方法
document.getElementById()
document.getElementsByTagName()

DOM扩展

document.getElementsByClassName()
  • 属性

以下属性均返回 HTMLCollection 对象的特殊集合

// 获取包含文档中所有带 name 特性的<a>元素
document.anchors
// 获取包含文档中所有带 href 特性的<a>元素
document.links
// 获取包含文档中所有的<img>元素
document.images
// 获取包含文档中所有的<form>元素
document.forms

# 2.5 document节点的文档写入

如果在文档加载结束(window.onload)后再调用，那么输出的内容将会重写整个页面

// 将文本写入到输出流中，write()会原样写入，而 writeln() 则会在字符串的末尾添加一个换行符（\n）
document.write(String);
document.writeIn(String);

# 三. Element类型节点

Node => Element => HTMLElement => [HTMLElement的子类型 =>] 各种 element 节点

# 3.1 element节点属性

element.nodeType：1
element.nodeName：对应标签名
element.nodeValue：null
element.parentNode：document 或 element
element.ownerDocument：document
// 特有
element.tagName: 对应标签名

# 3.2 element节点操作

  • 获取 element 节点
// 1.document 方法获取
document.getElementById()
document.getElementsByTagName()
// 2.element 也可以调用上述方法，此结果只会返回当前元素的后代
element.getElementById()
element.getElementsByTagName()

// 3.childNodes 属性, 需要判断节点类型 nodeType == 1
let arr = [];
for (let i = 0; i < element.childNodes; i++) {
    if (element.childNodes[i].nodeType == 1) {
       arr.push(element.childNodes[i]);
    }
}
console.log(arr);

DOM扩展

document.getElementsByClassName()
element.getElementsByClassName()

element.childElementCount
// 返回子元素（不包括文本节点和注释）的个数
element.firstElementChild
// 指向第一个子元素；firstChild 的元素版
element.lastElementChild
// 指向最后一个子元素；lastChild 的元素版
element.previousElementSibling
// 指向前一个同辈元素；previousSibling 的元素版
element.nextElementSibling
// 指向后一个同辈元素；nextSibling 的元素版

element.children
// 这个属性是 HTMLCollection 的实例，包含元素中是元素的子节点
  • 创建 element 节点
document.createElement()
// 这个方法只接收一个参数，即要创建元素的标签名
  • 增/改/删 element 节点参考通用节点操作

# 3.3 element节点的属性操作

// 查，增, 改，删
element.属性名

// 查
element.getAttribute(属性名)
// 改，增
element.setAttribute(属性名, 属性值)
// 删
element.removeAttribute(属性名)

通过 element.属性名 和 element.getAttribute(属性名) 获得的是属性的值，且两者相同 特例：style：通过 getAttribute()访问，返回的是 CSS 文本字符串，而通过属性来访问它则会返回一个对象 特例：事件：通过 getAttribute()访问，返回相应代码的字符串，而通过属性来访问它则会返回 JS 代码

DOM扩展

// dataset 属性是 DOMStringMap 的一个实例，也就是一个名值对儿的映射
// 每个 data-name 形式的属性都会有一个对应的没有 data-前缀的属性
element.dataset.属性名

// classList 属性是新集合类型 DOMTokenList 的实例
element.classList
// 查
element.classList[i]
element.classList.item(i)
element.classList.contains(属性名)
// 返回一个布尔值
// 增
element.classList.add(属性名)
// 删
element.classList.remove(属性名)
// 切换
element.classList.toggle(属性名)

# 四. Attr类型节点

Node => Attr => 各种 attribute 节点

# 4.1 attr节点的属性

attr.nodeType：2
attr.nodeName：对应属性名
attr.nodeValue：属性值
attr.parentNode：null
attr.ownerDocument：document
// 特有
attr.name: 对应属性名
attr.value: 属性值
// 用以区别特性是否为默认的: true，是默认的
attr.specified: Boolean

# 4.2 attr节点操作

  • 通常操作

并不建议直接访问属性节点

// 创建
document.createAttribute(属性名)
// 查
element.getAttributeNode(属性名)
// 增
element.setAttributeNode(属性名)
  • 通过元素的 attributes 属性
element.attributes
// 一个 attributes 属性包含一个 NamedNodeMap 对象，形式如下
NamedNodeMap {
    0: style, 
    1: onclick, 
    2: dataindex, 
    style: style, 
    onclick: onclick, 
    dataindex: dataindex, 
    length: 3
}

attributes 的方法不够方便，因此一般不在开发中使用

// 查
element.attributes[i]
element.attributes[属性名]
element.attributes.item(i)
element.attributes.getNamedItem(属性名)
// 增
element.attributes.setNamedItem(属性名)
// 删
element.attributes.removeNamedItem(属性名)

# 五. 其他DOM扩展

# 5.1 选择符API

querySelector(选择符)
querySelectorAll(选择符)
// 可以通过 Document 及 Element 类型的实例调用它们
element.matchesSelector(选择符)
// 如果调用元素与该选择符匹配，返回 true；否则，返回 false

# 5.2 焦点管理

元素获得焦点的方式有页面加载、用户输入和在代码中调用 focus()方法

document.activeElement
// 这个属性始终会引用 DOM 中当前获得了焦点的元素
document.hasFocus()
// 这个方法用于确定文档是否获得了焦点，返回一个布尔值

# 5.3 插入标记与插入文本

element.innerHTML
element.innerText
// 以下两个属性的作用会波及调用它们的元素本身
element.outerHTML
element.outerText

element.insertAdjacentHTML(插入位置, HTML字符串)
// 插入位置："beforebegin"，"afterbegin"，"beforeend"，"afterend"

# 5.4 滚动

// 让元素出现在视口中
// true： 或者不传入任何参数，那么窗口滚动之后会让调用元素的顶部与视口顶部尽可能平齐
// false：调用元素会尽可能全部出现在视口中
element.scrollIntoView(Boolean)
// 当前元素在视口中不可见的情况下，才滚动浏览器窗口或容器元素，最终让它可见
element.scrollIntoViewIfNeeded(Boolean)
