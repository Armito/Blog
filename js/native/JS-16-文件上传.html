# 文件上传

# 前端配置

技术需求:

  1. file 属性的 input
  2. FormData 内置对象
  3. Ajax
<input type="file" name="file" id="file_upload">

let file_upload = document.querySelector('#file_upload');
file_upload.onchange = function () {
    let formData = new FormData();
    formData.append('pic_test', this.files[0]);
    let xhr = new XMLHttpRequest();
    xhr.open('post', '/uploads');
    xhr.send(formData);
    xhr.onload = function () {
        console.log(xhr.responseText);
    }
};

# 后端配置

技术需求:

  1. express 框架
  2. express-formidable 包
const formidableMiddleware = require('express-formidable');
app.use(formidableMiddleware({
	// 文件上传目录
	uploadDir: path.join(__dirname, 'public', 'uploads'),
	// 最大上传文件为2M
	maxFileSize: 2 * 1024 * 1024,
	// 保留文件扩展名
	keepExtensions: true
}));
app.post('/uploads', async (req, res) => {
    // 建立结果数组
    let imgsPath = [];
    // console.log(req.files)
	// 如果用户上传了文件
	if (req.files) {
		// 循环结果对象
		for (let attr in req.files) {
			// 如果结果对象中存在path属性
			if(req.files[attr].path) {
				// 将值存储到结果数组中
				imgsPath.push({
					[attr]: req.files[attr].path.split('public')[1]
				});
			}
		}
	}
	// 将路径响应给客户端
	res.send(imgsPath);
});

# FormData 内置对象

FormData 内置对象用于异步上传文件, 属于 XMLHttpRequest 2级的内容

# 创建FormData对象

let formdata = new FormData();
let formdata = new FormData(file_upload);

# FormData对象的实例方法

formdata.set(key, value); // 会覆盖相同键名的属性值
formdata.get(key); // 返回指定键名的第一个属性值
formdata.getAll(key); // 返回指定键名的所有属性值
formdata.append(key, value); // 不会覆盖相同键名, 在属性后面继续添加
formdata.has(key); // 判断键名是否存在
formdata.delete(key); // 珊瑚指定键
