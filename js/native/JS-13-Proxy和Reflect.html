# Proxy & Reflect

# 一. Proxy

# 1.1 Proxy 的作用

Proxy 用作对象代理, 它会拦截对对象的操作, 对这些操作的默认行为进行过滤或修改

let o = {
    nickname: 'Pikachu'
};
let handler = {
    get: function (targetObj, key) {},
    set: function (targetObj, key, value) {},
    ...
};
// oProxy 就是 o 的代理
let oProxy = new Proxy(o, handler);

# 1.2 Proxy 可拦截的操作

get(targetObj, key, [receiver]): 属性读取操作; proxy.foo, proxy['foo'] set(targetObj, key, value, [receiver]): 属性赋值操作: proxy.foo = v, proxy['foo'] = v, 返回一个布尔值 has(targetObj, key): key in proxy 操作: 返回一个布尔值 deleteProperty(targetObj, key): delete proxy[key] 操作, 返回一个布尔值

# 1.3 Proxy 拦截属性读取操作应用

# 1.3.1 修改读取结果

let o = {
    nickname: 'Pikachu'
};
let oProxy = new Proxy(o, {
    get: function (targetObj, key) {
        if (key === 'nickname') {
            return 'Charizard';
        }
    }
});
oProxy.nickname; // 'Charizard'

# 1.3.2 读取失败报错

let o = {
    nickname: 'Pikachu'
};
let oProxy = new Proxy(o, {
    get: function (targetObj, key) {
        if (key in targetObj) {
            return targetObj[key];
        } else {
            throw new Error(`Property '${key}' does not exist`)
        }
    }
});
o.truename; // undefined
oProxy.truename; // error: Property 'truename' does not exist

# 1.3.3 链式读取属性

// 使用立即执行函数封装
let pipe =  (function () {
    // 被代理的对象, 用来存放属性, 这些属性均为对方法的引用 
    let funObj = {
        double: n => n * 2,
        pow: n => n * n
    };
    // 返回代理生成函数或结果
    return function (n) {
        // 存放链式调用的处理函数
        let funArr = [];
        // 生成代理
        let oProxy = new Proxy(funObj, {
            // get 处理函数, 调用 get 之外的属性均返回代理本身
            get: function (target, key) {
                if (key === 'get') {
                    // 返回处理结果, 使用数组的 reduce 方法处理
                    return funArr.reduce((acc, cur) => cur(acc), n);
                }
                // 否则将调用存放再数组中
                funArr.push(target[key]);
                // 返回代理以便链式调用
                return oProxy;
            }
        });
        // 第一次调用 pipe 时返回代理
        return oProxy;
    }
})()        
console.log(pipe(3).double.pow.get); // 36

# 1.4 Proxy 拦截属性赋值操作应用

# 1.4.1 数据验证

let oProxy = new Proxy(person = {}, {
    set: function (targetObj, key, value) {
        if (key === 'age') {
            if (!Number.isInteger(value)) {
                throw new Error(`Property 'age' must be a integer!`);
            } else {
                if (value > 200) {
                    throw new Error(`Property 'age' must be less than 200`);
                } else {
                    targetObj[key] = value;
                    return true;
                }
            }
        } else {
            targetObj[key] = value;
            return true;
        }
    }
})
oProxy.age = 'pikachu'; // Error: Property 'age' must be a integer!
oProxy.age = 28; // 28
oProxy.age = 283; // Error: Property 'age' must be less than 200
oProxy.hobby = 'code'; // code

# 1.4.2 对象私有属性保护

let oProxy = new Proxy(person = {
    _name: 'Pikachu'
}, {
    set: function (targetObj, key, value) {
        if (key[0] === '_') {
            throw new Error(`Property '${key}' is private!`);
        }
        targetObj[key] = value;
        return true;
    }
});
oProxy._name = 'Charizard'; // Error: Property '_name' is private!
oProxy.hobby = 'code'; // code

# 二. Reflect

# 2.1 Reflect 的作用

Reflect 是对对象属性操作的方法化

# 2.2 Reflect 保留的属性操作

Reflect 保留的对对象的属性操作方法与 Proxy 可拦截的操作一致, 且均为静态方法

# 2.3 Reflect.get(target, key, receiver)

# 2.3.1 对象属性读取

let o = {
    first: 1,
    second: 2,
    add: function () {
        return this.first + this.second;
    },
    get sub() {
        return this.second - this.first;
    }
};
let recObj = {
    first: 5,
    second: 2,
};
Reflect.get(o, 'add'); // ƒ () { return this.first + this.second; }
Reflect.get(o, 'sub'); // 1
// 如果对象方法的 name 属性设置了取值函数, 则 this 绑定 receiver
Reflect.get(o, 'sub', recObj); // -3

# 2.3.2 保留属性操作的默认行为

let oProxy = new Proxy(o = {
    nickname: 'Pikachu'
}, {
    get: function(targetObj, key) {
        console.log('get', targetObj[key]);
        return Reflect.get(targetObj, key);
    }
})
oProxy.nickname; // get Pikachu

# 2.4 Reflect.set(target, key, value, receiver)

# 2.4.1 对象属性赋值

let o = {
    set val(v) {
        return this.third = v;
    }
};
let recObj = {
    third: 3
};
Reflect.set(o, 'first', 1); // o.first: 1
Reflect.set(o, 'second', 2); // o.second: 2
// 如果对象方法的 name 属性设置了取值函数, 则 this 绑定 receiver
Reflect.set(o, 'third', 4, recObj);
recObj.third; // 4

# 2.4.2 Reflect.set 会触发 Proxy.defineProperty 拦截

let handler = { 
    set(target , key, value, receiver) { 
        console.log('set'); 
        Reflect.set(target, key, value, receiver);
    },
    defineProperty(target, key, descriptor) { 
        console.log('defineProperty'); 
        Reflect.defineProperty(target, key, descriptor) ; 
    }
};
let obj = new Proxy(p = {}, handler); 
obj.a = 'A'; 
// set 
// defineProperty
